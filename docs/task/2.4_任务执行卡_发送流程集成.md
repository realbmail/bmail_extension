## 一、任务目标

整合编辑器、收件人管理和加密功能,实现完整的邮件发送流程,通过 postMessage 将加密包传递给宿主页面的适配器。

## 二、技术方案

### 2.1 发送流程

```
用户点击发送
  → 验证收件人
  → 验证邮件内容
  → 执行加密
  → 构建发送消息
  → postMessage 发送到宿主
  → 等待宿主响应
  → 显示发送结果
```

### 2.2 消息协议

```typescript
// Iframe → 宿主

  };

```

### 2.3 UI 状态

- **编辑中**: 用户正在撰写邮件
- **验证中**: 检查收件人和内容
- **加密中**: 正在加密邮件内容
- **发送中**: 正在通过 postMessage 发送
- **成功**: 发送成功,显示确认
- **失败**: 发送失败,显示错误

## 三、实现步骤

### Step 1: 创建 sender.ts 模块

```typescript
// src/compose/sender.ts

import { Encryptor, EncryptedPackage } from './encryptor';

export interface SendOptions {
  recipients: string[];
  subject: string;
  content: string;
}

export interface SendResult {
  success: boolean;
  error?: string;
  messageId?: string;
}

export class Sender {
  private static readonly SEND_TIMEOUT = 30000; // 30秒超时

  /**
   * 发送邮件
   */
  static async send(options: SendOptions): Promise<SendResult> {
    try {
      // 1. 验证输入
      const validation = this.validateInput(options);
      if (!validation.valid) {
        return {
          success: false,
          error: validation.error
        };
      }

      // 2. 加密内容
      const encryptResult = await Encryptor.encrypt(
        options.content,
        options.recipients
      );

      if (!encryptResult.success || !encryptResult.encryptedPackage) {
        return {
          success: false,
          error: encryptResult.error || '加密失败'
        };
      }

      // 3. 构建消息
      const message = this.buildSendMessage(
        options.recipients,
        options.subject,
        encryptResult.encryptedPackage
      );

      // 4. 发送到宿主
      const result = await this.sendToHost(message);

      return result;
    } catch (error) {
      console.error('Send error:', error);
      return {
        success: false,
        error: error instanceof Error ? error.message : '发送失败'
      };
    }
  }

  /**
   * 验证输入
   */
  private static validateInput(options: SendOptions): { valid: boolean; error?: string } {
    if (!options.recipients || options.recipients.length === 0) {
      return { valid: false, error: '请至少添加一个收件人' };
    }

    if (!options.subject || options.subject.trim().length === 0) {
      return { valid: false, error: '请输入邮件主题' };
    }

    if (!options.content || options.content.trim().length === 0) {
      return { valid: false, error: '邮件内容不能为空' };
    }

    return { valid: true };
  }

  /**
   * 构建发送消息
   */
  private static buildSendMessage(
    recipients: string[],
    subject: string,
    encryptedPackage: EncryptedPackage
  ) {
    return {
      type: 'SEND_EMAIL',
      payload: {
        recipients,
        subject,
        encryptedContent: Encryptor.serializePackage(encryptedPackage),
        metadata: {
          timestamp: Date.now(),
          version: encryptedPackage.version
        }
      }
    };
  }

  /**
   * 发送消息到宿主页面
   */
  private static async sendToHost(message: any): Promise<SendResult> {
    return new Promise((resolve) => {
      // 设置超时
      const timeout = setTimeout(() => {
        window.removeEventListener('message', responseHandler);
        resolve({
          success: false,
          error: '发送超时,请重试'
        });
      }, this.SEND_TIMEOUT);

      // 监听响应
      const responseHandler = (event: MessageEvent) => {
        // 验证来源
        if (!event.origin.startsWith('chrome-extension://')) {
          return;
        }

        if (event.data.type === 'SEND_EMAIL_RESPONSE') {
          clearTimeout(timeout);
          window.removeEventListener('message', responseHandler);

          resolve({
            success: event.data.payload.success,
            error: event.data.payload.error,
            messageId: event.data.payload.messageId
          });
        }
      };

      window.addEventListener('message', responseHandler);

      // 发送消息
      window.parent.postMessage(message, '*');
      console.log('Sent SEND_EMAIL message to host');
    });
  }
}
```

### Step 3: 在 compose.ts 中集成发送流程

```typescript
// src/compose.ts
import { ComposeEditor } from './compose/editor';
import { RecipientManager } from './compose/recipient_manager';
import { Encryptor } from './compose/encryptor';
import { Sender, SendOptions } from './compose/sender';

let editor: ComposeEditor;
let recipientManager: RecipientManager;

/**
 * 显示发送状态
 */
function showSendStatus(type: 'loading' | 'success' | 'error', message: string) {
  const statusDiv = document.getElementById('send-status');
  if (!statusDiv) return;

  const statusContent = statusDiv.querySelector('.status-content');
  const statusMessage = statusDiv.querySelector('.status-message');

  if (statusContent && statusMessage) {
    statusContent.className = `status-content status-${type}`;
    statusMessage.textContent = message;
  }

  statusDiv.style.display = 'block';

  // 成功或失败后自动隐藏
  if (type === 'success' || type === 'error') {
    setTimeout(() => {
      statusDiv.style.display = 'none';
    }, 3000);
  }
}

/**
 * 隐藏发送状态
 */
function hideSendStatus() {
  const statusDiv = document.getElementById('send-status');
  if (statusDiv) {
    statusDiv.style.display = 'none';
  }
}

/**
 * 处理发送
 */
async function handleSend() {
  try {
    // 获取主题
    const subjectInput = document.getElementById('subject-input') as HTMLInputElement;
    const subject = subjectInput?.value.trim() || '';

    // 获取收件人
    const recipients = recipientManager.getValidEmails();

    // 获取内容
    const content = editor.getHTML();

    // 构建发送选项
    const options: SendOptions = {
      recipients,
      subject,
      content
    };

    // 禁用发送按钮
    const sendBtn = document.getElementById('send-btn') as HTMLButtonElement;
    if (sendBtn) {
      sendBtn.disabled = true;
    }

    // 显示发送中状态
    showSendStatus('loading', '正在发送邮件...');

    // 执行发送
    const result = await Sender.send(options);

    // 恢复发送按钮
    if (sendBtn) {
      sendBtn.disabled = false;
    }

    if (result.success) {
      showSendStatus('success', '邮件发送成功!');

      // 清空表单
      setTimeout(() => {
        clearForm();
      }, 2000);
    } else {
      showSendStatus('error', result.error || '发送失败');
    }
  } catch (error) {
    console.error('Send error:', error);
    showSendStatus('error', '发送过程中发生错误');

    // 恢复发送按钮
    const sendBtn = document.getElementById('send-btn') as HTMLButtonElement;
    if (sendBtn) {
      sendBtn.disabled = false;
    }
  }
}

/**
 * 处理取消
 */
function handleCancel() {
  // 发送关闭消息
  window.parent.postMessage({
    type: 'CLOSE_PANEL',
    payload: null
  }, '*');
}

/**
 * 清空表单
 */
function clearForm() {
  // 清空主题
  const subjectInput = document.getElementById('subject-input') as HTMLInputElement;
  if (subjectInput) {
    subjectInput.value = '';
  }

  // 清空编辑器
  editor.clear();

  // 清空收件人
  recipientManager.clear();
}

/**
 * 初始化按钮事件
 */
function initButtons() {
  const sendBtn = document.getElementById('send-btn');
  const cancelBtn = document.getElementById('cancel-btn');

  sendBtn?.addEventListener('click', handleSend);
  cancelBtn?.addEventListener('click', handleCancel);
}

// 页面加载完成后初始化
window.addEventListener('DOMContentLoaded', () => {
  initRecipientManager();
  initEditor();
  initButtons();

  // 通知宿主页面 Iframe 已就绪
  window.parent.postMessage({
    type: 'READY',
    payload: null
  }, '*');
});
```

## 六、风险与应对

### 风险 1: postMessage 通信失败
**影响**: 无法将加密包传递给宿主
**应对**:
- 添加超时机制
- 添加重试逻辑
- 提供友好的错误提示

### 风险 2: 宿主页面未响应
**影响**: 发送流程卡住
**应对**:
- 设置 30 秒超时
- 超时后提示用户
- 允许用户重试

### 风险 3: 表单状态管理复杂
**影响**: 可能出现状态不一致
**应对**:
- 使用清晰的状态机
- 在关键节点验证状态
- 添加详细的日志
